const CSV_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSnV2YkleLyZu7oH4sRkaSV1k1KToH05LgWDi4xR0wcKquNeuI8134CcF3n8O1OlbvAIedrQq4XvS4f/pub?output=csv'; async function fetchPrice() { try { const response = await fetch( CSV_URL + '&t=' + Date.now() ); const text = await response.text(); const rows = text.trim().split('\n'); let currentPrice = null; let previousClose = null; rows.forEach(row => { const columns = row.split(','); const symbol = columns[0] ?.trim() .toUpperCase(); if ( symbol.includes('BANK') ) { currentPrice = parseFloat(columns[1]); previousClose = parseFloat(columns[2]); } }); if ( currentPrice === null || previousClose === null ) { return; } const priceElement = document.getElementById('price'); priceElement.innerText = currentPrice.toLocaleString( 'en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 } ); if ( currentPrice > previousClose ) { priceElement.style.color = '#16a34a'; } else if ( currentPrice < previousClose ) { priceElement.style.color = '#dc2626'; } else { priceElement.style.color = '#111111'; } } catch (e) { console.error(e); document.getElementById('price').innerText = 'Unavailable'; } }
