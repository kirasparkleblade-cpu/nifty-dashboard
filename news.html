async function loadNews() { try { const response = await fetch( RSS_URL + encodeURIComponent( 'https://news.google.com/rss/search?q=' + '(NIFTY OR BANKNIFTY OR NSE OR Sensex OR RBI OR inflation OR oil OR Fed OR earnings OR global markets OR economy OR Nasdaq OR Dow Jones OR crude oil OR geopolitics OR US markets)&hl=en-IN&gl=IN&ceid=IN:en' ) + '&t=' + Date.now() ); const data = await response.json(); const newsList = document.getElementById( 'newsList' ); newsList.innerHTML = ''; if ( !data.items ) { newsList.innerHTML = `
Unable to load news
`; return; } let count = 0; data.items.forEach(item => { if (count >= 12) return; if ( !isRecent(item.pubDate) ) return; const card = document.createElement('a'); card.href = item.link; card.target = '_blank'; card.className = 'news-card'; card.innerHTML = `
${item.title}
${formatDate( item.pubDate )}
NEWS
`; newsList.appendChild(card); count++; }); if (count === 0) { newsList.innerHTML = `
No recent news available
`; } } catch (error) { console.error(error); document.getElementById( 'newsList' ).innerHTML = `
Unable to load news
`; } }
